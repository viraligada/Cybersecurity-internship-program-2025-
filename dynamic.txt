  <script>
    const correctPin = "1234"; // The correct PIN
    let enteredPin = "";
    const pinDisplay = document.getElementById("pinDisplay");
    const keypadContainer = document.getElementById("keypad");
    const backspaceBtn = document.getElementById("backspace");
    const clearBtn = document.getElementById("clear");

    function shuffleArray(array) {
      for (let i = array.length - 1; i > 0; i--) {
        const j = Math.floor(Math.random() * (i + 1));
        [array[i], array[j]] = [array[j], array[i]];
      }
    }

    function createShuffledKeypad() {
      keypadContainer.innerHTML = "";
      const digits = Array.from({ length: 10 }, (_, i) => i);
      shuffleArray(digits);

      digits.forEach((digit) => {
        const btn = document.createElement("button");
        btn.textContent = digit;
        btn.addEventListener("click", () => {
          enteredPin += digit;
          updateDisplay();
          if (enteredPin.length === correctPin.length) {
            checkPin();
          }
        });
        keypadContainer.appendChild(btn);
      });
    }

    function updateDisplay() {
      pinDisplay.textContent = "•".repeat(enteredPin.length);
    }

    function checkPin() {
      if (enteredPin === correctPin) {
        alert("Password was correct");
        reset();
      } else {
        alert("Password is invalid");
        reset();
      }
    }

    function reset() {
      enteredPin = "";
      updateDisplay();
      createShuffledKeypad();
    }

    backspaceBtn.addEventListener("click", () => {
      enteredPin = enteredPin.slice(0, -1);
      updateDisplay();
    });

    clearBtn.addEventListener("click", () => {
      reset();
    });

    // Initialize keypad on page load
    createShuffledKeypad();
  </script>
